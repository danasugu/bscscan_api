Future<EtherScanAbiModel> getAbi({
  required String? address,
}) async {
  const module = 'contract';
  const action = 'getabi';

  var query = {
    'module': module,
    'action': action,
    'address': address,
    'apiKey': apiKey
  };

  return (await get(query)).fold(
    (l) => EtherScanAbiModel.empty(),
    (r) => EtherScanAbiModel.fromJson(r),
  );
}
